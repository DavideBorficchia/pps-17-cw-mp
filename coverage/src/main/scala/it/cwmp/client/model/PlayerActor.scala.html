55   override def preStart(): Unit = {
56     log.info(s"Initializing the game-view actor...")
57     gameViewActor = system.actorOf(Props[GameViewActor], "game-view")
58     log.info(s"Subscribing to cluster changes...")
59     cluster.subscribe(self, initialStateMode = InitialStateAsEvents,
60       classOf[MemberEvent], classOf[UnreachableMember])
61   }

63   override def postStop(): Unit = cluster.unsubscribe(self)
64 
65   override def receive: Receive = clusterBehaviour orElse lobbyBehaviour
66 
67   private def clusterBehaviour: Receive = {
68     case MemberUp(member) =>
69       log.info("Member is Up: {}", member.address)
70       log.debug("Cluster size: " + cluster.state.members.size)
71       if (cluster.state.members.size == roomSize) enterGame
72     case UnreachableMember(member) =>
73       log.info("Member detected as unreachable: {}", member)
74     case MemberRemoved(member, previousStatus) =>
75       log.info(
76         "Member is Removed: {} after {}",
77         member.address, previousStatus)
78     case _: MemberEvent => // ignore
79   }
80 
81   private def backToLobby: Unit = context.become(clusterBehaviour orElse lobbyBehaviour)
82 
83   private def lobbyBehaviour: Receive = {
84     case RetrieveAddress =>
85       sender() ! RetrieveAddressResponse(getAddress)
86     case StartGame(participants) =>
87       connectTo(participants)
88       roomSize = participants.size
89   }
90 
91   private def enterGame: Unit = {
92     context.become(clusterBehaviour orElse inGameBehaviour)
93     gameViewActor ! ShowGUI
94     gameViewActor ! NewWorld(GameMain.debugWorld)
95   }
96 
97   private def inGameBehaviour: Receive = distributedState.stateBehaviour orElse {
98     case EndGame => backToLobby
99   }
100 
101   private def connectTo(participants: List[Address]): Unit = {
102     cluster.join(AddressFromURIString(participants.head.address))
103   }
104 
105   private def getAddress: String = cluster.selfAddress + self.path.toString.substring(self.path.address.toString.length)
